175  private IEnumerable<IGameRecord> GetMultipleByQuery(string sql, object param)
176  {
177  return this.backingDatabase.Query<IEnumerable<IGameRecord>>(dbConnection =>
178  {
179  using (var query = dbConnection.QueryMultiple(sql, param))
180  {
181  try
182  {
183  var games = query.Read().Select(game => new Guid(game.uuid));
184 
185  var files = query.Read().Select(file => new
186  {
187  Guid = new Guid(file.uuid),
188  Game = new Guid(file.game),
189  Path = file.path,
190  MimeType = file.mimetype
191  });
192 
193  var metadatas = query.Read().Select(metadata => new
194  {
195  Guid = new Guid(metadata.uuid),
196  Record = new Guid(metadata.record),
197  Key = metadata.key,
198  Value = metadata.value
199  }).Select(m => new RecordMetadata(m.Guid, m.Record, m.Key, m.Value)
200  as IRecordMetadata);
201 
202  var fileRecords = (from f in files
203  let md = (from m in metadatas where m.Record == f.Guid select m)
204  .ToDictionary(md => md.Key, md => md)
205  select new FileRecord(f.Game, md, f.Path, f.MimeType)).ToList();
206  return (from game in games
207  let gameFiles =
208  (from f in fileRecords where f.Record == game select f as IFileRecord).ToList()
209  let md = (from m in metadatas where m.Record == game select m)
210  .ToDictionary(md => md.Key, md => md)
211  select new GameRecord(game, md, gameFiles)).ToList();
212  }
213  catch (SqliteException)
214  {
215  return new List<IGameRecord>();
216  }
217 
218  }
219  });
220  }
