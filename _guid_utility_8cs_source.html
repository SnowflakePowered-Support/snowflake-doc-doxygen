34  public static Guid Create(Guid namespaceId, string name, int version)
35  {
36  if (name == null)
37  throw new ArgumentNullException("name");
38  if (version != 3 && version != 5)
39  throw new ArgumentOutOfRangeException("version", "version must be either 3 or 5.");
40 
41  // convert the name to a sequence of octets (as defined by the standard or conventions of its namespace) (step 3)
42  // ASSUME: UTF-8 encoding is always appropriate
43  byte[] nameBytes = Encoding.UTF8.GetBytes(name);
44 
45  // convert the namespace UUID to network order (step 3)
46  byte[] namespaceBytes = namespaceId.ToByteArray();
47  GuidUtility.SwapByteOrder(namespaceBytes);
48 
49  // comput the hash of the name space ID concatenated with the name (step 4)
50  byte[] hash;
51  using (var algorithm = version == 3 ? (HashAlgorithm)MD5.Create() : (HashAlgorithm)SHA1.Create())
52  using(var incrementalHash = version == 3 ? IncrementalHash.CreateHash(HashAlgorithmName.MD5) : IncrementalHash.CreateHash(HashAlgorithmName.SHA1))
53  {
54  incrementalHash.AppendData(namespaceBytes);
55  incrementalHash.AppendData(nameBytes);
56  hash = incrementalHash.GetHashAndReset();
57  /*algorithm.TransformBlock(namespaceBytes, 0, namespaceBytes.Length, null, 0);
58  algorithm.TransformFinalBlock(nameBytes, 0, nameBytes.Length);
59  hash = algorithm.Hash;*/ //todo verify correctness;
60  }
61 
62  // most bytes from the hash are copied straight to the bytes of the new GUID (steps 5-7, 9, 11-12)
63  byte[] newGuid = new byte[16];
64  Array.Copy(hash, 0, newGuid, 0, 16);
65 
66  // set the four most significant bits (bits 12 through 15) of the time_hi_and_version field to the appropriate 4-bit version number from Section 4.1.3 (step 8)
67  newGuid[6] = (byte)((newGuid[6] & 0x0F) | (version << 4));
68 
69  // set the two most significant bits (bits 6 and 7) of the clock_seq_hi_and_reserved to zero and one, respectively (step 10)
70  newGuid[8] = (byte)((newGuid[8] & 0x3F) | 0x80);
71 
72  // convert the resulting UUID to local byte order (step 13)
73  GuidUtility.SwapByteOrder(newGuid);
74  return new Guid(newGuid);
75  }
