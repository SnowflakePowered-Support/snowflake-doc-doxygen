105  private IFileRecord GetZipStreamInformation(string zipFile)
106  {
107  using (ZipArchive archive = new ZipArchive(File.OpenRead(zipFile), ZipArchiveMode.Read))
108  {
109  var firstEntry = (from entry in archive.Entries.AsParallel()
110  let extension = Path.GetExtension(entry.FullName)?.ToLowerInvariant()
111  where this.validFileExtensions.Contains(extension)
112  //hopefully this doesn't come back to bite me in the ass.
113  where extension != ".zip"
114  select entry).FirstOrDefault();
115  if (firstEntry == null) return null;
116  Stream deflatedStream = new MemoryStream();
117  using (var stream = firstEntry.Open())
118  {
119  stream.CopyTo(deflatedStream);
120  }
121  IFileRecord romRecord = this.GetFileInformation(firstEntry.FullName, deflatedStream);
122  IFileRecord zipRecord = new FileRecord(zipFile, $"{romRecord.MimeType}+zip");
123  zipRecord.Metadata.Add(romRecord.Metadata);
124  zipRecord.Metadata.Add(FileMetadataKeys.RomZipRunnableFilename, firstEntry.FullName);
125  return zipRecord;
126  }

130  private IFileRecord GetFileInformation(string romfile, Stream romStream)
131  {
132  string ext = Path.GetExtension(romfile)?.ToLowerInvariant();
133  var potentialMimetypes = (from mimetype in this.validMimetypes
134  where mimetype.Contains(ext)
135  select mimetype.Key).ToList();
136 
137  if (!potentialMimetypes.Any()) return null;
138 
139  var romFileInfo = this.fileSignatures.GetInfo(romfile, romStream);
140  if (romFileInfo == null) throw new InvalidDataException("Unable to ascertain ROM type with given signatures.");
141  var fileInfo = romFileInfo?.Serial == null ?
142  this.GetFromHash(romfile, romFileInfo, romStream)
143  : this.GetFromSerial(romfile, romFileInfo, romStream);
144  romStream.Dispose();
145  return fileInfo;
146  }

165  private IFileRecord GetFromHash(string romfile, IRomFileInfo romFileInfo, Stream romStream)
166  {
167  string hash = FileHash.GetMD5(romStream); //timeout for large files
168  var romInfo = this.shiragameProvider.GetFromMd5(hash);
169  if (romInfo == null) return this.GuessFromFile(romfile, romFileInfo);
170  IFileRecord record = new FileRecord(romfile, romInfo.MimeType);
171  if (romFileInfo?.InternalName != null)
172  record.Metadata.Add(FileMetadataKeys.RomInternalName, romFileInfo.InternalName);
173  record.Metadata.Add(FileMetadataKeys.RomRegion, romInfo.Region);
174  record.Metadata.Add(FileMetadataKeys.FileHashMd5, romInfo.MD5);
175  record.Metadata.Add(FileMetadataKeys.FileHashSha1, romInfo.SHA1);
176  record.Metadata.Add(FileMetadataKeys.FileHashCrc32, romInfo.CRC32);
177  record.Metadata.Add(FileMetadataKeys.RomCanonicalTitle, new StructuredFilename(romInfo.FileName).Title);
178  record.Metadata.Add(FileMetadataKeys.RomPlatform, romInfo.PlatformId);

182  private IFileRecord GetFromSerial(string romfile, IRomFileInfo romInfo, Stream romStream)
183  {
184  var gamePlatform =
185  this.stoneProvider.Platforms.First(p => p.Value.FileTypes.Values.Contains(romInfo.Mimetype)).Value;
186  string romSerial = romInfo.Serial;
187  string mimeType = romInfo.Mimetype;
188  IFileRecord record = new FileRecord(romfile, $"{mimeType}");
189  if(romInfo.InternalName != null)
190  record.Metadata.Add(FileMetadataKeys.RomInternalName, romInfo.InternalName);
191  record.Metadata.Add(FileMetadataKeys.RomSerial, romSerial);
192  var serialInfo = this.shiragameProvider.GetFromSerial(gamePlatform.PlatformID, romSerial);
193  if (serialInfo == null)
194  return this.GetFromHash(romfile, romInfo, romStream); //fallback to hash
195  record.Metadata.Add(FileMetadataKeys.RomCanonicalTitle, new StructuredFilename(serialInfo.Title).Title);
196  record.Metadata.Add(FileMetadataKeys.RomRegion, serialInfo.Region);
197  record.Metadata.Add(FileMetadataKeys.RomPlatform, gamePlatform.PlatformID);
198  return record;
