65  private string ParseRegion()
66  {
67  var tagData = Regex.Matches(this.OriginalFilename,
68  @"(\()([^)]+)(\))");
69  var validMatch = (from Match tagMatch in tagData
70  let match = tagMatch.Groups[2].Value
71  from regionCode in (from regionCode in match.Split(',', '-') select regionCode.Trim())
72  where regionCode.Length != 2 || regionCode.ToLower().ToTitleCase() != regionCode
73  let isoRegion = StructuredFilename.ConvertToRegionCode(regionCode.ToUpperInvariant())
74  where isoRegion.Item1 != null
